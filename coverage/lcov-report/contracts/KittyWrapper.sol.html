pragma solidity 0.6.12;
 
import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
 
contract KittyWrapper is ERC721("Wrapped CryptoKitty", "wCK") {
    IERC721 constant public KITTY = IERC721(0x06012c8cf97BEaD5deAe237070F9587f8E7A266d);
 
    constructor() public {
        _setBaseURI("https://public.api.cryptokitties.co/v1/kitties/");
    }
 
    function wrap(uint256 _tokenId) public {
        _mint(msg.sender, _tokenId);
        KITTY.transferFrom(msg.sender, address(this), _tokenId);
    }
 
    function wrapMany(uint256[] calldata _tokenIds) external {
        for (uint256 i = 0; i < _tokenIds.length; i++)
            wrap(_tokenIds[i]);
    }
 
    function unwrap(uint256 _tokenId) public {
        require(ownerOf(_tokenId) == msg.sender);
        _burn(_tokenId);
        KITTY.safeTransferFrom(address(this), msg.sender, _tokenId);
    }
 
    function unwrapMany(uint256[] calldata _tokenIds) external {
        for (uint256 i = 0; i < _tokenIds.length; i++)
            unwrap(_tokenIds[i]);
    }
 
    function onERC721Received(address _from, uint256 _tokenId, bytes calldata _data) external returns (bytes4) {
        require(msg.sender == address(KITTY));
        _mint(_from, _tokenId);
        return KittyWrapper.onERC721Received.selector;
    }
}
